# 目录结构

```bash
endpoints/
├── academic/           # 学术相关
│   ├── courses.py     # 课程管理
│   ├── exams.py      # 考试管理
│   ├── grades.py     # 成绩管理
│   └── evaluations.py # 教学评价
├── auth/              # 认证授权
│   ├── auth.py       # 认证
│   └── rbac.py       # 权限控制
├── user/              # 用户管理
│   ├── users.py      # 用户管理
│   ├── students.py   # 学生管理
│   └── teachers.py   # 教师管理
├── system/            # 系统管理
│   ├── config.py     # 配置管理
│   ├── logs.py       # 日志管理
│   └── monitor.py    # 系统监控
├── tools/             # 工具
│   ├── files.py      # 文件管理
│   └── tools.py      # 通用工具
└── common/            # 公共模块
    ├── stats.py      # 统计
    └── health.py     # 健康检查
```
